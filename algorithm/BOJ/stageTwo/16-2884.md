import Foundation

// 1. 60분 기준으로 모든걸 바꿔준다
// 2. 45분 빼준다
// 3. 다시 시간, 분으로 바꿔준다.
// 4. 예외처리, min이 음수이면 60 - min해줌, hour도 23더해줌
let input = readLine()!
let inputArr = input.split(separator: " ")

let hour = Int(inputArr[0])!
let min = Int(inputArr[1])!
var sum: Int
var updatedHour: Int
var updatedMin: Int
sum = hour * 60 + min
sum -= 45
updatedHour = sum / 60
updatedMin = sum - updatedHour * 60
if (updatedMin < 0){
    updatedMin += 60
    updatedHour += 23
}
print("\(updatedHour) \(updatedMin)")
